Examples: Forms

Form Builder

Forms are now easy to build using JSON and the SWAM Form Builder! This can be used via a SWAM.Views.Form, a SWAM.Dialogs.showForm, or directly using SWAM.Form.build!

Form
Code
PORTAL.Pages.SimplePage = SWAM.Page.extend(SWAM.Ext.BS).extend({
  on_init: function() {
    this.appendChild("myform", new SWAM.Form.View({
        fields:[
            {
                label: "Full Name",
                name: "full_name",
                placeholder: "Enter Your Full Name"
            },
            {
                label: "Bio",
                name: "bio",
                type: "textarea",
                placeholder: "Tell us something about yourself"
            },
            {
                label: "Gender",
                name: "gender",
                type: "select",
                options: [
                    {
                        label:"Select Gender",
                        value: ""
                    },
                    "male",
                    "female",
                    "other"
                ]
            }
        ]
    }));
  },

  on_input_change: function(evt) {
      SWAM.toast(evt.name, evt.value);
  }
});
