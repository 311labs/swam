// OUR PAGE

PORTAL.Pages.MyPage = SWAM.Page.extend({
    template: ".pages.mypage",
    classes: "page-view page-padded has-topbar",

   on_init: function() {
    // add a chart view
    // this will look for an element with the id barchart in the pages html
    this.addChild("barchart", new SWAM.Views.Chart({type:"bar"});
    // add data to the view
    this.children.barchart.setLabels(SWAM.DataSets.months.slice(0,6));
    this.children.barchart.addDataSet("beers", [10, 20, 30, 40, 50, 60]);
    this.children.barchart.addDataSet("shots", [2, 80, 20, 15, 40, 50]);

    }
});

// OUR APP

PORTAL.App = SWAM.App.extend({

  // Pages are added to your app in the apps on_init_pages
  on_init_pages: function() {
    // param 1 is the page id / name
    // param 2 is an instance of the page
    // param 3 is your url routing
    this.addPage("swam_views", new PORTAL.Pages.ExampleViews(), ["swam/views"]);
  }

});
